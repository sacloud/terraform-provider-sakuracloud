スイッチ(sakuracloud_switch)

設定例

resource sakuracloud_switch "myswitch" {
  name = "sw01"
  #bridge_id = "${sakuracloud_bridge.br.id}"
  description = "Switch from terraform for SAKURA CLOUD"
  tags        = ["tag1", "tag2"]
}
