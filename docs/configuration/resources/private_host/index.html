専有ホスト(sakuracloud_private_host)

設定例

resource "sakuracloud_private_host" "private_host" {
    name = "private_host"
    description = "PrivateHost from terraform for SAKURA CLOUD"
    tags = ["tag1" , "tag2"]
}
