パケットフィルタ(sakuracloud_packet_filter)

設定例

resource "sakuracloud_packet_filter" "myfilter" {
    name = "myfilter"
    description = "PacketFilter from terraform for SAKURA CLOUD"
    expressions = {
        protocol = "tcp"
        source_nw = "192.168.2.0/24"
        source_port = "0-65535"
        dest_port = "80"
        allow = true
    }
    expressions = {
        protocol = "ip"
        source_nw = "0.0.0.0"
        allow = false
        description = "Deny all"
    }
}
